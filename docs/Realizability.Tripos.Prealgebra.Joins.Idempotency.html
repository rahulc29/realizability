  private PredicateX = Predicate {ℓ'' = ℓ''} X

  x⊔x≤x : ∀ x → x ⊔ x ≤ x
  x⊔x≤x x =
      return
        (pr₂ ,
          (λ x' a a⊩x⊔x →
            transport
              (propTruncIdempotent (x .isPropValued x' (pr₂ ⨾ a)))
              (do
                a⊩x⊔x ← a⊩x⊔x
                let
                  goal : ((pr₁ ⨾ a ≡ k) × (pr₂ ⨾ a) ⊩ ∣ x ∣ x') ⊎ ((pr₁ ⨾ a ≡ k') × (pr₂ ⨾ a) ⊩ ∣ x ∣ x') → (pr₂ ⨾ a) ⊩ ∣ x ∣ x'
                  goal = λ { (inl (_ , pr₂a⊩x)) → pr₂a⊩x ; (inr (_ , pr₂a⊩x)) → pr₂a⊩x }
                return (goal a⊩x⊔x))))

  x≤x⊔x : ∀ x → x ≤ x ⊔ x

    let prover = ` pair ̇ ` true ̇ # fzero in
    ∣ λ* prover ,
